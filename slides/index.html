loop


					let mut counter = 1;
					let loop_ret = loop {
					  counter += 1;

					  if counter == 15 {
					    break counter
					  }
					};
					assert_eq!(loop_ret, 15); // used in unit tests

					let loop_ret = loop {
					  break
					};
					println!("loop_ret: {:?}", loop_ret);

impl Message {
  fn get_error_code(&self) -> Option<&u32> {
    match self {
      Message::Ok (_) => None,
      Message::Skipped => None,
      Message::Error { code, .. } => Some(code)
    }
  }
}
